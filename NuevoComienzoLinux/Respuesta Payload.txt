	function validarRespuestas(){ 
	 	 var validacion =false;
		 var numPreguntas = document.forma.yTotalPreguntas.value;	
		// alert(numPreguntas);	 
		 for(var k=1; k<=numPreguntas; k++){		 
		   var tipo_pregunta = document.getElementById("yTipoPregunta"+k).value;	
		   var tam_res =  document.getElementById("yTamRespuestas"+k).value;	
		   var nombre = "yRespuestaPregunta"+k;  
		   var clave_pregunta = findObj ("yCvePregunta"+k);
		                       
		  if(tipo_pregunta == 2 || tipo_pregunta == 5){
		     if(validarSimple(k,tam_res))
			  { validacion =true; }
			 else
			  {  return false;
				 validacion =false;
			  } 
		  }
		  
		  if(tipo_pregunta == 4 || tipo_pregunta == 1){
		     if(validarMultiple(k,tam_res))
			 { validacion = true;  }
			 else
			 {
			    return false;
				validacion = false
			 }
		  }
		  
		  if(tipo_pregunta == 3)
		  {
			 if(validarAbierta(k))
			  { validacion =true; }
			 else
			  {  return false;
				 validacion =false;
			  } 
		  }
		 }
	
	     return validacion;
	}
	
	function validarSimple(pregunta, respuestas){
	   var validarS = false;
	   for(var j = 1; j <= respuestas; j++){
	    if(document.getElementById("yRespuestaPregunta"+pregunta+""+j).checked)
		{
		  if(document.getElementById("yResAbierta"+pregunta+""+j).value == 1)  
		  {
		     var respuesta_abierta = findObj("yRespuestaAbierta"+pregunta+""+j); 
			 if(Trim( respuesta_abierta.value ) != "" && Trim( respuesta_abierta.value ) != "-")
			 {
			   validarS=true
			   return validarS;
			 }
			 else
			 {		 
			   alert( "Es necesario indicar la información para la respuesta  de la pregunta No. " + pregunta );
			   var respuesta = findObj("yRespuestaAbierta"+pregunta+""+j);
			   respuesta.focus();
			   return false;
			   validarS=false;
			 }
		  }
		  else{
		    validarS = true;
			return validarS;
		  } 
		}	   
	   }
	   
	   if(validarS==false)
	    alert("Falta responder la pregunta No. "+(pregunta));
	   
	   return validarS;
	}
	
	function validarAbierta(pregunta){

	}
	
	function validarMultiple(pregunta, respuestas){
	
	   var validarM = false;
	   var checado = 0;
	   for(var j = 1; j <= respuestas; j++){
	    if(document.getElementById("#"+pregunta+""+j).checked)
		{
		  if(document.getElementById("yResAbierta"+pregunta+""+j).value == 1)  
		  {
		     var respuesta_abierta = findObj("yRespuestaAbierta"+pregunta+""+j); 
			 if(Trim( respuesta_abierta.value ) != "" && Trim( respuesta_abierta.value ) != "-")
			 {
			    checado++;
			 }
			 else
			 {		 
			   alert( "Es necesario indicar la información para la respuesta  de la pregunta No. " + pregunta );
			   var respuesta = findObj("yRespuestaAbierta"+pregunta+""+j);
			   respuesta.focus();
			   return false;
			   validarM=false;
			 }
		  }
		  else{
		  checado++;
		  } 
		}	   
	   }
	   
	   if(checado>0){ validarM=true }
	   else
	   {
	    alert("Falta responder la pregunta No. "+(pregunta));
	   }
	   
	   return validarM;
	}
